  - NP
    - N
      - Alex
  - VP
    - V
      - eats
    - N
      - soup
  - ADV
    - everyday

- S
  - NP
    - N
      - children
  - VP
    - V

- S
  - NP
    - N
      - cats
  - VP
    - V
      - adore
    - NP
      - ADJ
        - salty
      - N
        - soup

Rules:<br>
S -> NP VP | NP VP ADV<br>
<br>
NP -> N | ADJ N<br>
VP -> V NP<br>
<br>
N -> "Alex" | "soup" | "children" | "cats" | "chips"<br>
V -> " eats" | "like" | "adore"<br>
ADJ -> "salty"<br>
ADV -> "everyday"
